-- Create database NS

Create table Moderator (
	ID integer,
	Naam varchar(255),
	primary key(ID));

Create table Zuil (
	ID varchar(255),
	Locatie varchar(255),
	primary key(ID));

Create table Bericht (
 	ID serial,
 	Naam varchar(255),
 	Datum_Bericht date,
	tijd_bericht time,
 	Bericht varchar(255),
 	Status	integer,
 	opmerking varchar(255),
 	Datum_Moderatie date,
	tijd_moderatie time,
 	ZuilID varchar(255),
 	ModeratorID integer,
 	primary key (ID),
 	foreign key (ZuilID) references Zuil (ID),
 	foreign key (ModeratorID) references Moderator (ID));
